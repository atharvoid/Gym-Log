enum _Range { all, m1, m6, y1 }

extension on _Range {
  String get label => switch (this) { _Range.all => 'All Time', _Range.m1 => '1M', _Range.m6 => '6M', _Range.y1 => '1Y' };
  DateTime? since() {
    final now = DateTime.now();
    return switch (this) {
      _Range.all => null,
      _Range.m1 => now.subtract(const Duration(days: 30)),
      _Range.m6 => now.subtract(const Duration(days: 182)),
      _Range.y1 => now.subtract(const Duration(days: 365)),
    };
  }
}
