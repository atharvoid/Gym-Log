Future<List<DailyVolumeSample>> dailyVolumeForRoutine(String routineId) {

    '''
    SELECT
      DATE(ws.started_at) AS day,
      SUM(ws.total_volume_kg) AS volume
    FROM workout_sessions ws
    WHERE ws.routine_id = ?
    GROUP BY DATE(ws.started_at)

    ''',
